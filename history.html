/******************************************** * FIRESTORE INIT ********************************************/ var db = firebase.firestore(); /******************************************** * CONSTANTS ********************************************/ const QUARTERS = [ "Q425","Q126","Q226","Q326","Q426", "Q127","Q227","Q327","Q427" ]; /******************************************** * HELPERS ********************************************/ const el = id => document.getElementById(id); function escapeHtml(str) { if (!str) return ""; return String(str) .replace(/&/g,"&") .replace(//g,">"); } /******************************************** * LOAD QUARTER + WEEK DATA ********************************************/ async function loadHistoryForQuarter(q) { el("milestones-history").innerHTML = "Loading…"; el("weekly-history").innerHTML = "Loading…"; el("resourcing-history").innerHTML = "Loading…"; el("daily-history").innerHTML = "Loading…"; try { const msSnap = await db.doc("dashboard/milestones").get(); const wpSnap = await db.doc("dashboard/weeklyPlans").get(); const dlSnap = await db.doc("dashboard/dailyLogs").get(); const qrSnap = await db.doc("dashboard/resourcing").get(); const msData = msSnap.exists ? msSnap.data() : {}; const wpData = wpSnap.exists ? wpSnap.data() : {}; const dailyData = dlSnap.exists ? dlSnap.data() : {}; const qrData = qrSnap.exists ? qrSnap.data() : {}; renderMilestonesHistory(q, msData); renderWeeklyDropdown(q, wpData); renderWeeklyHistory(q, wpData); renderResourcingHistory(q, qrData); renderDailyHistory(dailyData); } catch (err) { console.error(err); el("milestones-history").innerHTML = "Error loading history."; } } /******************************************** * MILESTONES HISTORY ********************************************/ function renderMilestonesHistory(q, data) { const container = el("milestones-history"); const qdata = data[q]; if (!qdata || Object.keys(qdata).length === 0) { container.innerHTML = "

No milestones recorded for this quarter.

"; return; } let html = ""; Object.keys(qdata).forEach(cat => { html += `

${cat}

    `; qdata[cat].forEach(m => { html += `
  • ${escapeHtml(m.title)}
    Date: ${escapeHtml(m.date || "—")}
    Personnel: ${escapeHtml(m.people || "—")}
    Progress: ${m.progress || 0}%
    • `; }); html += "
"; }); container.innerHTML = html; } /******************************************** * WEEKLY — POPULATE WEEK DROPDOWN ********************************************/ function renderWeeklyDropdown(q, data) { const weekSelect = el("history-week-select"); weekSelect.innerHTML = ""; const qdata = data[q]; if (!qdata) return; // week keys = "2025-W46" const weeks = Object.keys(qdata).sort(); weeks.forEach(w => { const opt = document.createElement("option"); opt.value = w; opt.textContent = w; weekSelect.appendChild(opt); }); if (weeks.length > 0) { weekSelect.value = weeks[weeks.length - 1]; // latest week default } } /******************************************** * WEEKLY HISTORY ********************************************/ function renderWeeklyHistory(q, data) { const container = el("weekly-history"); const weekSelect = el("history-week-select"); const selectedWeek = weekSelect.value; const qdata = data[q]; if (!qdata || !qdata[selectedWeek]) { container.innerHTML = "

No weekly plans saved for this week.

"; return; } const weekObj = qdata[selectedWeek]; let html = `

${selectedWeek}

`; Object.keys(weekObj).forEach(cat => { html += `

${cat}

    `; weekObj[cat].forEach(t => { html += `
  • ${escapeHtml(t.title)} — ${t.person} (${t.percent}%)
    ${t.subtasks.map(s => `
    • ${escapeHtml(s)}
    `).join("")}
    • `; }); html += "
"; }); container.innerHTML = html; } /******************************************** * QUARTERLY RESOURCING HISTORY ********************************************/ function renderResourcingHistory(q, data) { const container = el("resourcing-history"); const qdata = data[q]; if (!qdata) { container.innerHTML = "

No resourcing saved.

"; return; } let html = ""; Object.keys(qdata).forEach(cat => { html += `

${cat}

    `; Object.keys(qdata[cat]).forEach(person => { html += `
  • ${person}: ${qdata[cat][person]}%
    • `; }); html += "
"; }); container.innerHTML = html; } /******************************************** * DAILY LOG HISTORY ********************************************/ function renderDailyHistory(dailyData) { const container = el("daily-history"); if (!dailyData || Object.keys(dailyData).length === 0) { container.innerHTML = "

No daily logs saved.

"; return; } let html = ""; Object.keys(dailyData).sort().forEach(day => { html += `

${day}

    `; const people = dailyData[day]; Object.keys(people).forEach(name => { html += `
  • ${name}
    ${escapeHtml(people[name].today || "—")}
    • `; }); html += "
"; }); container.innerHTML = html; } /******************************************** * POPULATE QUARTER SELECT ********************************************/ function initQuarterSelect() { const qsel = el("history-quarter-select"); QUARTERS.forEach(q => { const opt = document.createElement("option"); opt.value = q; opt.textContent = q; qsel.appendChild(opt); }); qsel.value = "Q425"; // default match dashboard } /******************************************** * EVENT LISTENERS ********************************************/ document.getElementById("history-quarter-select").addEventListener("change", e => { loadHistoryForQuarter(e.target.value); }); document.getElementById("history-week-select").addEventListener("change", () => { const q = el("history-quarter-select").value; db.doc("dashboard/weeklyPlans") .get() .then(snap => { const data = snap.exists ? snap.data() : {}; renderWeeklyHistory(q, data); }); }); /******************************************** * INIT ********************************************/ initQuarterSelect(); loadHistoryForQuarter("Q425");
